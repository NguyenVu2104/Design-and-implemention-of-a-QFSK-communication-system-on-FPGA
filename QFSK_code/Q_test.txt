module data_source (
    input logic clk_i,          // Clock input (50 MHz)
    input logic rst_ni,         // Active-low reset
    output logic [31:0] fsw     // Frequency selection word (32-bit)
);
    // Define constants
    localparam SYMBOL_DURATION = 33350; // 33,350 cycles = 0.667 ms at 50 MHz
    localparam TEXT_LENGTH = 1800;        // Example text length in bits (adjust as needed)

    // Memory array to store the text as bits
    logic text_memory [0:TEXT_LENGTH-1]; // Changed to memory array
    
    // Address counter for accessing text memory (in bits)
    logic [$clog2(TEXT_LENGTH)-1:0] bit_addr;
    
    // Timer to control symbol duration
    logic [$clog2(SYMBOL_DURATION)-1:0] symbol_timer;

    // Current symbol being output
    logic [1:0] current_symbol;

    // Load example text into memory at initialization
    initial begin
        $readmemb("E:/symbols.txt", text_memory); // Read into memory array
    end

    // Sequential logic to manage timing and symbol output
    always_ff @(posedge clk_i or negedge rst_ni) begin
        if (!rst_ni) begin
            symbol_timer <= '0;
            bit_addr <= '0;
            current_symbol <= 2'b00;
        end else begin
            if (symbol_timer < SYMBOL_DURATION - 1) begin
                symbol_timer <= symbol_timer + 1;
            end else begin
                symbol_timer <= '0;
                // Extract the next 2 bits from the text
                if (bit_addr < TEXT_LENGTH - 1) begin
                    current_symbol <= {text_memory[bit_addr], text_memory[bit_addr+1]}; // Extract 2 bits
                    bit_addr <= bit_addr + 2;
                end else begin
                    current_symbol <= {text_memory[1], text_memory[0]}; // Loop back to start
                    bit_addr <= 2; // Skip first 2 bits
                end
            end
        end
    end

    // Combinational logic to map 2-bit symbol to 32-bit fsw
    always_comb begin
        case (current_symbol)
            2'b00: fsw = 32'd1030792;  // 12 kHz
            2'b01: fsw = 32'd1546188;  // 18 kHz
            2'b10: fsw = 32'd2061584;  // 24 kHz
            2'b11: fsw = 32'd2576980;  // 30 kHz
            default: fsw = 32'd0; // Default to 12 kHz
        endcase
    end
endmodule

module dds (
    input logic clk_i,           // System clock
    input logic rst_ni,           // Reset signal
    input logic [31:0] fsw,    // Frequency Control

    output logic signed [15:0] sine_out     // 16-bit sine wave amplitude from LUT (signed)
);

    // Internal signals
    logic [9:0] phase_out;     // 10-bit phase output from Phase Accumulator

    // Phase Accumulator Module
    phase_accumulator pa (
        .clk_i(clk_i),
        .rst_ni(rst_ni),
        .fsw(fsw),
        .phase_out(phase_out)
    );

    // Phase-to-Amplitude Module (Sine LUT)
    phase_to_amplitude pta (
        .clk_i(clk_i),
        .phase_in(phase_out),
        .sine_out(sine_out)
    );

endmodule

// Phase Accumulator Module
module phase_accumulator (
    input logic clk_i,
    input logic rst_ni,
    input logic [31:0] fsw,
    output logic [9:0] phase_out
);
    logic [31:0] phase_acc;

    always_ff @(posedge clk_i or negedge rst_ni) begin
        if (!rst_ni) begin
            phase_acc <= 32'd0;
            phase_out <= 10'd0;
        end else begin
            phase_acc <= phase_acc + fsw;
            phase_out <= phase_acc[31:22]; // Extract 10 MSBs for LUT addressing
        end
    end
endmodule

module channel (
    input logic clk_i,                    // Clock input
    input logic rst_ni,                   // Active-low reset
    input logic signed [15:0] signal_in,  // Input signal (e.g., sine_out from DDS)
    output logic signed [16:0] signal_out // Output signal with added noise
);
    // Parameters
    localparam LFSR_MIN_VALUE = 12'd2000; // Minimum LFSR value

    // Internal signals
    logic [11:0] lfsr_reg;                // LFSR register (12-bit)
    logic [11:0] lfsr_constrained;        // LFSR value constrained to be >= 2000

    // LFSR for pseudo-random noise generation (12-bit)
    always_ff @(posedge clk_i or negedge rst_ni) begin
        if (!rst_ni) begin
            lfsr_reg <= 12'hACE;          // Initial seed (0xACE = 2766, above 2000)
        end else begin
            // Feedback polynomial: x^12 + x^11 + x^10 + x^4 + 1
            lfsr_reg <= {lfsr_reg[10:0], lfsr_reg[11] ^ lfsr_reg[10] ^ lfsr_reg[9] ^ lfsr_reg[3]};
        end
    end

    // Constrain LFSR value to be >= 2000
    always_comb begin
        if (lfsr_reg < LFSR_MIN_VALUE) begin
            lfsr_constrained = LFSR_MIN_VALUE;
        end else begin
            lfsr_constrained = lfsr_reg;
        end
    end

    // Add noise to the input signal
    always_ff @(posedge clk_i) begin
        if (!rst_ni) begin
            signal_out <= 17'd0;
        end else begin
            // Directly add LFSR value as noise (signal_out is 21-bit to accommodate the wider range)
            signal_out <= $signed({{1{signal_in[15]}}, signal_in}) + $signed({{5{lfsr_constrained[11]}}, lfsr_constrained});
        end
    end
endmodule

module qfsk (
    input logic clk_i,                    // Clock input (23.04 MHz)
    input logic rst_ni,                   // Active-low reset
    output logic [31:0] fsw,              // Frequency selection word from data_source
    output logic signed [15:0] sine_out,  // Sine wave output from DDS
    output logic signed [16:0] signal_with_noise, // Signal with noise (16-bit for testbench)
    output logic [1:0] original_symbol,   // Original symbol from data_source
    output logic [1:0] recovered_symbol,  // Recovered symbol from decision
);

    // Instantiate submodules
    data_source ds (
        .clk_i(clk_i),
        .rst_ni(rst_ni),
        .fsw(fsw)
    );
    dds dds_inst (
        .clk_i(clk_i),
        .rst_ni(rst_ni),
        .fsw(fsw),
        .sine_out(sine_out)
    );
    channel chan (
        .clk_i(clk_i),
        .rst_ni(rst_ni),
        .signal_in(sine_out),
        .signal_out(signal_with_noise)
    );

    // Assign original_symbol from data_source
    assign original_symbol = ds.current_symbol;
endmodule
